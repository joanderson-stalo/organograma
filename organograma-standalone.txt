    <script>
        // Dados completos omitidos para economizar espaço
        // Veja o arquivo completo no repositório

        const departments = [
            {
                id: 'products',
                title: 'Produtos',
                manager: { name: 'Vaga aberta', role: 'Gestor de Produtos - PM', vacant: true },
                team: [
                    { name: 'Stephany', role: 'Product Owner - PO' },
                    { name: 'Cainã Sena', role: 'Especialista em Implementação Técnica' },
                ],
                description: 'PMO, cronogramas e governança de produtos',
                color: 'teal',
            },
            // ... outros departamentos
        ];

        function renderDepartments() {
            const container = document.getElementById('departments');
            departments.forEach(dept => {
                const teamCount = dept.team.length + (dept.manager ? 1 : 0);
                const vacantCount = dept.team.filter(p => p.vacant).length + (dept.manager?.vacant ? 1 : 0);

                const card = `
                    <div class="dept-wrapper">
                        <div class="line-v-small"></div>
                        <div class="dept-card color-${dept.color}" onclick="showDetail('${dept.id}')">
                            <div class="dept-header">
                                <h3 class="dept-title">${dept.title}</h3>
                                <svg class="chevron" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"/>
                                </svg>
                            </div>
                            ${dept.manager ? `
                                <div class="dept-manager">
                                    <p class="manager-name ${dept.manager.vacant ? 'vacant' : ''}">${dept.manager.name}</p>
                                </div>
                            ` : ''}
                            <div class="dept-info">
                                <svg class="icon-users" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 4.354a4 4 0 110 5.292M15 21H3v-1a6 6 0 0112 0v1zm0 0h6v-1a6 6 0 00-9-5.197M13 7a4 4 0 11-8 0 4 4 0 018 0z"/>
                                </svg>
                                <span>${teamCount} pessoas</span>
                            </div>
                            ${vacantCount > 0 ? `<div class="vacant-count">${vacantCount} vaga${vacantCount > 1 ? 's' : ''}</div>` : ''}
                        </div>
                    </div>
                `;
                container.innerHTML += card;
            });
        }

        function showDetail(id) {
            // Lógica de navegação para página de detalhes
            document.getElementById('homeView').classList.add('hidden');
            document.getElementById('detailView').classList.add('active');
        }

        function hideDetail() {
            document.getElementById('homeView').classList.remove('hidden');
            document.getElementById('detailView').classList.remove('active');
        }

        // Initialize
        renderDepartments();
    </script>
